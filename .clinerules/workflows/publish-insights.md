```
python3 scripts/publish_report.py --file "./reports/Initiative-Insights.md" --title "Initiative Insights"

python3 scripts/publish_report.py --file "./reports/Sprint-Insights.md" --title "Sprint Insights"

python3 scripts/publish_report.py --file "./reports/Team-Insights.md" --title "Team Insights" 
```
